CREATE TABLE `t_finance_gir_dt` (
  `id` varchar(255) NOT NULL DEFAULT '',
  `create_by` varchar(255) DEFAULT NULL,
  `create_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP,
  `del_flag` tinyint(4) DEFAULT '0',
  `update_by` varchar(255) DEFAULT NULL,
  `update_time` datetime NOT NULL DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,
  `remark` varchar(255) NOT NULL DEFAULT '' COMMENT '备注',
  `main_id` varchar(255) NOT NULL COMMENT '主表id',
  `by_time` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '联运外放买量汇总月份',
  `sum_time` datetime DEFAULT NULL ON UPDATE CURRENT_TIMESTAMP COMMENT '收款时间-现金流业务日期',
  `amount` decimal(18,2) NOT NULL DEFAULT '0.00' COMMENT '收款金额',
  PRIMARY KEY (`id`),
  KEY `fgir_dt_index` (`main_id`),
  CONSTRAINT `fgir_dt_index` FOREIGN KEY (`main_id`) REFERENCES `t_finance_game_intermodal_release` (`id`) ON DELETE CASCADE ON UPDATE CASCADE
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='联运外放买量汇总收款';
